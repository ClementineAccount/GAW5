Layout
	use default




Buttons
	*fishingButton
        tag:button
		name:Go Fishing
		desc:Attempt to fish... maybe you'd get a catch! If not, you have to try again after a bit of time.
		on click:anim icon wobble
		on click:hookState is 1
		on click:hide fishingButton
		on click:
			if (chance(50%)) 
				if (chance(50%))
					if (chance(40%))
						yield 1 blueFish
					else
						yield 1 yellowFish
					end
				else
					yield 1 redFish
				end
			end
		end
		icon:https://media.githubusercontent.com/media/ClementineAccount/GAW5/main/fish.png
		no text
		class:bigButton hasFlares
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red

	*sellFishButton
		name:Sell Fish
		desc:Sell fishes for money.
		on click:
			yield (redFish) money
			redFish is 0
		end
		on click:
			yield ((blueFish) * 5) money
			blueFish is 0
		end
		on click:
			yield ((yellowFish) * 10) money
			yellowFish is 0
		end
		text:Sell fish at the market.

	*timeTicker
		desc:Increases the timer when the fishingButton is clicked
		hidden
		on tick:if (hookState is 1) yield 1 timeUnit
		on tick:
			if (timeUnit >= 2) 
				show fishingButton
				hookState is 0
				timeUnit is 0
			end
		end
