Layout
  use default



Buttons
	*fishingButton
        tag:button
		name:Go Fishing
		desc:Attempt to fish... maybe you'd get a catch! If not, you have to try again after a bit of time.
		on click:anim icon wobble
		on click:hookState is 1
		on click:hide fishingButton
		on catch:
			if (chance(50%)) 
				if (chance(50%))
					if (chance(40%))
						yield 1 blueFish
					else
						yield 1 yellowFish
					end
				else
					yield 1 redFish
				end
			end
		end
		on click:
			if (chance(50%)) 
				if (chance(50%))
					if (chance(40%))
						yield 1 blueFish
					else
						yield 1 yellowFish
					end
				else
					yield 1 redFish
				end
			end
		end
		icon:https://media.githubusercontent.com/media/ClementineAccount/GAW5/main/fish.png
		no text
		class:bigButton hasFlares
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red

	*flavorText
		name:flavor text item
		desc:Quirky flavor text that comments on things just like in Cookie Clicker
		text:A nice day for fishing...
		on tick:hide flavorText
		no tooltip
		no click
		class:flavor-text

	*sellFishButton
		name:Sell Fish
		desc:Sell fishes for money.
		on click:
			yield (redFish * (redFishPrice)) money
			redFish is 0
		end
		on click:
			yield ((blueFish) * (blueFishPrice)) money
			blueFish is 0
		end
		on click:
			yield ((yellowFish) * (yellowFishPrice)) money
			yellowFish is 0
		end
		on tick:
			hide sellFishButton
			if (redFish > 0)
				show sellFishButton
			else if (blueFish > 0)
				show sellFishButton
			else if (yellowFish > 0)
				show sellFishButton 
			end
		end
		text:Sell fish at the market.
		tooltip origin:bottom
		tooltip class:red



	*timeTicker
		desc:Increases the timer when the fishingButton is clicked
		hidden
		on tick:if (hookState is 1) yield 1 timeUnit
		on tick:
			if (timeUnit >= 2) 
				show fishingButton
				hookState is 0
				timeUnit is 0
			end
		end
